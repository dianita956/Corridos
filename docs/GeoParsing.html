def geojson(shape, utm, label=None):
    gdf_ = gpd.GeoDataFrame(geometry=[shape], crs=utm).to_crs("epsg:4386")
    if label:
        gdf_ = gdf_.assign(label=label)
    return gdf_.__geo_interface__
